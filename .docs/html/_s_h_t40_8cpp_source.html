71uint8_t SHT40::measure(SHT40_meas_t type)

84uint8_t SHT40::temperature(int16_t& out)
85{
86 // If temperature data is zero
87 if (!mData.temp) return SHT40_NOK;
88
89 // Calculate temperature from SHT40 data in desired temperature unit
90 if (getUnit() == SHT40_UNIT_C) out = -45 + (175 * ((mData.temp[0] << 8) | mData.temp[1]) / 655) / 100;
91 else out = -49 + (315 * (((mData.temp[0] << 8) | mData.temp[1]) / 655)) / 100;
92
93 // Reset temperature data to zero
94 mData.temp[0] = 0;
95 mData.temp[1] = 0;
96 return SHT40_OK;
97}

99uint8_t SHT40::rh(uint8_t& out)
100{
101 // If RH data is zero
102 if (!mData.rh) return SHT40_NOK;
103
104 // Calculate RH from SHT40 data
105 out = -6 + (125 * (((mData.rh[0] << 8) | mData.rh[1]) / 655)) / 100;
106 if (out > 100) out = 100;
107 else if (out < 0) out = 0;
108
109 // Reset RH data to zero
110 mData.rh[0] = 0;
111 mData.rh[1] = 0;
112 return SHT40_OK;
113}

115uint32_t SHT40::whoAmI(void)
116{
117 uint8_t cmd = SHT40_SN;
118
119 // Send S/N read command
120 I2CWrite(address, (uint8_t*)&cmd, 1, 1);
121
122 // Read S/N from SHT40
123 I2CRead(address, (uint8_t*)&snData, 6, 0);
124
125 // If NAK is received
126 if (header == SHT40_NAK) return SHT40_NOK;
127
128 // Concat 4x8-bit S/N data to 32-bit S/N
129 return ((snData.sn1[1] << 24) | (snData.sn1[0] << 16) | (snData.sn2[1] << 8) | snData.sn2[0]);
130}

140uint8_t SHT40::init(SHT40_unit_t u)
141{
142 // If external read and/or write methods are not provided
143 if (!I2CRead) return SHT40_I2CH_R;
144 if (!I2CWrite) return SHT40_I2CH_W;
145
146 // Set temperature unit
147 setUnit(u);
148
149 // Read SHT40's S/N
150 uint32_t tmp = whoAmI();
151
152 // If S/N is bits are not zeros or ones
153 if (tmp == 0 || tmp == 0xFFFFFFFF) return SHT40_NOK;
154
155 return SHT40_OK;
156}

178static uint16_t getDelay(SHT40_meas_t type)
179{
180 switch (type)
181 {
182 case TRH_H: return 9; // 8.2ms
183 case TRH_M: return 5; // 4.5ms
184 case TRH_L: return 2; // 1.7ms
185
186 case TRH_H_H02W1S:
187 case TRH_H_H011W1S:
188 case TRH_H_H002W1S: return 1150; // 1100ms
189
190 case TRH_H_H02W01S:
191 case TRH_H_H011W01S:
192 case TRH_H_H002W01S: return 120; // 110ms
193
194 default: break;
195 }
196
197 return 1;
198}
