72uint8_t SHT40::measure(SHT40_meas_t type)
73{
74 // Write command to SHT40
75 I2CWrite(address, (uint8_t*)&type, 1, getMeasureDelay(type));
76
77 // Read from SHT40
78 I2CRead(address, (uint8_t*)&mData, 6, 0);
79
80 // If NAK is received
81 if (header == SHT40_NAK) return SHT40_NOK;
82
83 // Clear old data flags
84 controlReg &= ~(SHT40_OLD_RH | SHT40_OLD_TEMP);
85
86 return SHT40_OK;
87}

89uint8_t SHT40::temperature(int8_t& out)
90{
91 uint8_t ret = SHT40_OK;
92
93 // If temperature data is zero or old
94 if (!mData.temp[0] && !mData.temp[1]) return SHT40_NOK;
95 if (controlReg & SHT40_OLD_TEMP) ret = SHT40_OLD_DATA;
96
97 // Calculate temperature from SHT40 data in desired temperature unit
98 if (getUnit() == SHT40_UNIT_C) out = -45 + (175 * ((mData.temp[0] << 8) | mData.temp[1]) / 655) / 100;
99 else out = -49 + (315 * (((mData.temp[0] << 8) | mData.temp[1]) / 655)) / 100;
100
101 // Set old data flag for temperature
102 controlReg |= SHT40_OLD_TEMP;
103 return ret;
104}

106uint8_t SHT40::rh(uint8_t& out)
107{
108 uint8_t ret = SHT40_OK;
109
110 // If RH data is zero or old
111 if (!mData.rh[0] && !mData.rh[1]) return SHT40_NOK;
112 if (controlReg & SHT40_OLD_RH) ret = SHT40_OLD_DATA;
113
114 // Calculate RH from SHT40 data
115 out = -6 + (125 * (((mData.rh[0] << 8) | mData.rh[1]) / 655)) / 100;
116
117 // Limit output value between 0 and 100
118 if (out > 100) out = 100;
119 else if (out < 0) out = 0;
120
121 // Set old data flag FOR RH
122 controlReg |= SHT40_OLD_RH;
123 return ret;
124}

126uint32_t SHT40::whoAmI(void)
127{
128 uint8_t cmd = SHT40_SN;
129
130 // Send S/N read command
131 I2CWrite(address, (uint8_t*)&cmd, 1, 1);
132
133 // Read S/N from SHT40
134 I2CRead(address, (uint8_t*)&snData, 6, 0);
135
136 // If NAK is received
137 if (header == SHT40_NAK) return SHT40_NOK;
138
139 // Concat 2x16-bit S/N data to 32-bit S/N
140 return ((snData.sn1 << 16) | snData.sn2);
141}

151uint8_t SHT40::init(SHT40_unit_t u)
152{
153 // If external read and/or write methods are not provided
154 if (!I2CRead) return SHT40_I2CH_R;
155 if (!I2CWrite) return SHT40_I2CH_W;
156
157 // Set temperature unit
158 setUnit(u);
159
160 // Read SHT40's S/N
161 uint32_t tmp = whoAmI();
162
163 // If S/N is bits are not zeros or ones
164 if (!tmp || tmp == 0xFFFFFFFF) return SHT40_NOK;
165
166 return SHT40_OK;
167}

193static uint16_t getMeasureDelay(SHT40_meas_t type)
194{
195 switch (type)
196 {
197 case TRH_H: return 9; // 8.2ms
198 case TRH_M: return 5; // 4.5ms
199 case TRH_L: return 2; // 1.7ms
200
201 case TRH_H_H02W1S:
202 case TRH_H_H011W1S:
203 case TRH_H_H002W1S: return 1150; // 1100ms
204
205 case TRH_H_H02W01S:
206 case TRH_H_H011W01S:
207 case TRH_H_H002W01S: return 120; // 110ms
208
209 default: break;
210 }
211
212 return 1;
213}
