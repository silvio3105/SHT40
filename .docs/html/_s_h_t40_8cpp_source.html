77uint8_t SHT40::temperature(uint32_t& out)
78{
79 if (!data.mData.temp) return SHT40_NOK;
80
81 if (getUnit() == SHT40_UNIT_C) out = -45 + (175 * ((uint32_t)data.mData.temp / 65535));
82 else out = -49 + (315 * ((uint32_t)data.mData.temp / 65535));
83 data.mData.temp = 0;
84 return SHT40_OK;
85}

87uint8_t SHT40::rh(uint32_t& out)
88{
89 if (!data.mData.rh) return SHT40_NOK;
90
91 out = -6 + (125 * ((uint32_t)data.mData.rh / 65535));
92 if (out > 100) out = 100;
93 else if (out < 0) out = 0;
94
95 data.mData.rh = 0;
96 return SHT40_OK;
97}

99uint32_t SHT40::whoAmI(void)
100{
101 uint8_t cmd = SHT40_SN;
102
103 I2CWrite(address, (uint8_t*)&cmd, 1);
104 I2CRead(address, (uint8_t*)&data.snData, 6);
105
106 if (data.header == SHT40_NAK) return SHT40_NOK;
107
108 return (uint32_t)((data.snData.sn1 << 16) | data.snData.sn2);
109}

117uint8_t SHT40::init(void)
118{
119 if (!I2CRead) return SHT40_I2CH_R;
120 if (!I2CWrite) return SHT40_I2CH_W;
121
122 uint32_t tmp = whoAmI();
123 if (!tmp || tmp == 0xFFFFFFFF) return SHT40_NOK;
124
125 return SHT40_OK;
126}
